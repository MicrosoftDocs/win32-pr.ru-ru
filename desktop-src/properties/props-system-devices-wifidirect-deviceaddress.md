```
propertyDescription
   name = System.Devices.WiFiDirect.DeviceAddress
   shellPKey = PKEY_Devices_WiFiDirect_DeviceAddress
   formatID = 1506935D-E3E7-450F-8637-82233EBE5F6E
   propID = 13
```
